<script>

  // Центр Москвы

  const map = L.map('map').setView([55.7558, 37.6173], 12);


  // Подключаем карту OpenStreetMap

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

    attribution: '&copy; OpenStreetMap contributors'

  }).addTo(map);


  // Пример перемещения — линия из точки А в точку B

  const from = [55.7558, 37.6173]; // Красная площадь

  const to = [55.7600, 37.6250];   // Арбат


  L.polyline([from, to], { color: 'blue' }).addTo(map)

    .bindPopup('Перемещение учителя');

</script>
